void BinaryTreeTest::test_node_creation_with_value() {
  test_header( "node creation with value" );

  int test_value = 42;
  Node node( test_value );

  print_test( node.data == test_value );
}

void BinaryTreeTest::test_node_children_are_null_by_default() {
  test_header( "node children are null by default" );

  Node node;

  print_test( node.left == NULL && node.right == NULL );
}

void BinaryTreeTest::test_tree_contains() {
  test_header( "tree insert and contains" );

  int something = 42;
  Tree tree;
  tree.insert( something );

  print_test( tree.contains( something ) == true );
}

void BinaryTreeTest::test_tree_remove() {
  test_header( "tree remove" );

  int something = 42;
  Tree tree;
  tree.insert( something );
  tree.remove( something );

  print_test( tree.contains( something ) == false );
}

Sample Code: tree.cpp

This sample includes the removeMatch method. This removes a node from the binary tree while maintaining the binary tree structure. The node argument is a pointer to the node to be removed. The parent argment is a pointer to the parent node of node. left is true if node is the left child of parent. There are four cases to consider: node has no children, node only has a right child, node only has a left child, and node has both left and right children. In each case, node is properly deleted and the tree is relinked appropriately.

template 
void Tree::removeMatch( Node* parent, Node* node, bool left ){
  if ( root != NULL ){
    Node* trash;
    T smallestInRight;

    // no children
    if ( node->left == NULL && node->right == NULL ){
      trash = node;
      if ( left ){
        parent->left = NULL;
      }
      
      else{
        parent->right = NULL;
      }

      delete trash;
    }

    // right child only
    else if ( node->left == NULL && node->right != NULL ){
      if ( left ){
        parent->left = node->right;
      }

      else{
        parent->right = node->right;
      }

      node->right = NULL;
      trash = node; 
      delete trash;
    }

    //left child only
    else if ( node->left != NULL && node->right == NULL ){
      if ( left ){
        parent->left = node->left;
      }

      else{
        parent->right = node->left;
      }

      node->left = NULL;
      trash = node; 
      delete trash;
    }

    // two children
    else{
      smallestInRight = getMinimumValue( node->right );
      removeHelp( node, smallestInRight );
      node->data = smallestInRight;
    }
  }
}
